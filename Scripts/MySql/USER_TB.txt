  PASSWORDLASTUPDATEDATE DATETIME,
  LASTSUCCESSLOGINDATE DATETIME,
  LASTFAILEDLOGINDATE DATETIME,
  MODIFIEDBY BIGINT(20) UNSIGNED,
  DATEMODIFIED DATETIME,
